Line Graph

var data = {
	'labels' : ['Jan', 'Feb', 'Mar','Apr', 'May', 'Jun', 'Jul'],
	'datasets' : [
		{
			strokeColor : "rgba(220,220,220,1)",
			pointColor : "#000",
			data : [65,59,90,81,56,55,40]
		},
		{
			strokeColor : "rgba(151,187,205,1)",
			pointColor : "#00ff00",
			data : [20,48,40,12,96,27,110]
		}
	]
};
options = {
	bg_color: '#eee'
};
$('.line').miniGraph("line", data, options);

Default options

options =  {
	width : 600,
	height : 400,
	bg_color : '#eee',
	gridColor : '#ccc',
	baizerCurve : true,
	showLabels : true,
	showGrid : true,
	showDots : true,
	textSize : 14,
	textColor: '#00ff00'
};

Pie Graph

var data =[
	{
		value: 30,
		color:"#F38630"
	},
	{
		value : 50,
		color : "#E0E4CC"
	},
	{
		value : 100,
		color : "#69D2E7"
	}
];
options = {
	bg_color: '#eee'
};
$('.pie').miniGraph("pie", data, options);

Default options

options =  {
	width : 600,
	height : 400,
	bg_color : '#eee',
	showStroke : true,
	showLabels : true,
	strokeWidth : 3,
	strokeColor : '#fff',
	textSize : 15,
	textColor: '#000'		
};
